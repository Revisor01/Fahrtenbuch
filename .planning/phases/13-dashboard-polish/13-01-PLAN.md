  ```js
    const result = Array(12).fill(0);
    fahrten.forEach(f => {
      if (!f.datum) return;
      const [y, m] = f.datum.split('-');
      if (parseInt(y) === statistikJahr) {
        result[parseInt(m) - 1]++;
      }
    });
    return result;
  }, [fahrten, statistikJahr]);
  ```

- Responsive: Auf Mobile bleibt das Layout einspaltig, keine Aenderung noetig da FahrtForm bereits responsive ist
