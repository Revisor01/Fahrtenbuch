```javascript
const getMonthLabel = (yearMonth) => {
  const [y, m] = yearMonth.split('-');
  return new Date(parseInt(y), parseInt(m) - 1).toLocaleString('de-DE', { month: 'short' });
};
```

**Zeitraum-Rendering (isZeitraum === true):**
Statt eines einzelnen Status-Badge eine kompakte Chip-Zeile rendern. Fuer jeden Monats-Key im verschachtelten Objekt `summary.abrechnungsStatus?.[key]` einen Chip anzeigen:

```jsx
{isZeitraum ? (
  <div className="flex flex-wrap gap-1">
    {Object.entries(summary.abrechnungsStatus?.[key] || {})
      .sort(([a], [b]) => a.localeCompare(b))
      .map(([monthKey, statusData]) => {
        const label = getMonthLabel(monthKey);
        if (statusData?.erhalten_am) {
          return (
            <span key={monthKey} className="status-badge-primary text-xs">
              <CheckCircle2 size={12} />
              <span>{label}</span>
            </span>
          );
        } else if (statusData?.eingereicht_am) {
          return (
            <span key={monthKey} className="status-badge-warning text-xs">
              <Circle size={12} />
              <span>{label}</span>
            </span>
          );
        } else {
          return (
            <span key={monthKey} className="status-badge-secondary text-xs">
              <AlertCircle size={12} />
              <span>{label}</span>
            </span>
          );
        }
      })}
  </div>
) : (
  /* bestehende flache Status-Logik hier beibehalten */
)}
```

1. **Hilfsfunktion fuer offene Monate pro Traeger (EX-01):** Eine Funktion erstellen, die im Zeitraum-Modus die offenen Monate fuer einen Traeger ermittelt:

```javascript
const getOffeneMonateRange = (traegerKey) => {
  if (!isZeitraum) return null; // Im Einzelmonat kein Filtern
  const traegerStatus = summary.abrechnungsStatus?.[traegerKey];
  if (!traegerStatus || typeof traegerStatus !== 'object') return null;
  // Filtere Monate, die NICHT eingereicht und NICHT erhalten sind
  return Object.entries(traegerStatus)
    .filter(([monthKey, statusData]) => !statusData?.eingereicht_am && !statusData?.erhalten_am)
    .map(([monthKey]) => monthKey);
};
```

2. **Hilfsfunktion: Hat Traeger offene Monate? (EX-02):**
```javascript
const hatOffeneMonate = (traegerKey) => {
  if (!isZeitraum) return true; // Im Einzelmonat immer anzeigen
  const offene = getOffeneMonateRange(traegerKey);
  return offene === null || offene.length > 0;
};
```
